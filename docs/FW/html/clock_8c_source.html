266 uint32_t unix_time_in_seconds(uint8_t sec, uint8_t min, uint8_t hrs,
267  uint8_t day, uint8_t mon, uint16_t year) {
268 
269  uint32_t ts = 0;
270 
271  // Add up the seconds from all prev years, up until this year.
272  uint8_t years = 0;
273  uint8_t leap_years = 0;
274  for (uint16_t y_k = EPOCH_YEAR; y_k < year; y_k++) {
275  if (IS_LEAP_YEAR(y_k)) {
276  leap_years++;
277  } else {
278  years++;
279  }
280  }
281  ts += ((years * days_per_year[0]) + (leap_years * days_per_year[1]))
282  * SEC_PER_DAY;
283 
284  // Add up the seconds from all prev days this year, up until today.
285  uint8_t year_index = (IS_LEAP_YEAR(year)) ? 1 : 0;
286  for (uint8_t mo_k = 0; mo_k < (mon - 1); mo_k++) { // days from previous months this year
287  ts += days_per_month[year_index][mo_k] * SEC_PER_DAY;
288  }
289  ts += (day - 1) * SEC_PER_DAY; // days from this month
290 
291  // Calculate seconds elapsed just today.
292  ts += hrs * SEC_PER_HOUR;
293  ts += min * SEC_PER_MIN;
294  ts += sec;
295 
296  return ts;
297 }
